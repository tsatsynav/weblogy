Задание 3: необходимо создать функцию getDivisorsCount(), которая будет высчитывать количество делителей для переданного числа. Если он не передан или значение было не числом, то из функции необходимо вернуть NaN.

Также если number меньше нуля либо является не целым (можно проверить при помощи функции Number.isInteger(number)), то выводите в модальное окно при помощи alert() сообщение “number должен быть целым числом и больше нуля!”.

Код:


      let inputNumber = +(prompt("Введите число:").trim());

      function getDivisorsCount(number) {
        if (!Number(number)) {
          return NaN;
        } else if (number < 0 || number % 1 !== 0) {
          return `Число должно быть целым и больше нуля!`;
        }
        
        let count = 0;
        let i = 1;
        for (i; i <= number; i++) {
            if (number % i === 0) {
            count += 1;
          } 
        }
        return count;    
      }

      alert(getDivisorsCount(inputNumber));
